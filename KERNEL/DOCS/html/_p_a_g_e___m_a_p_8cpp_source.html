35 bool GlobalPageMap::AllocatePAgeMap()
36 {
37 size_t Ram = SYSTEM::SYSTEM_INFO::SystemInfo::GetInstance().installedRam;
38 if (Ram == 0) {
39 return false; // RAM not initialized
40 }
41 constexpr size_t pageSize = EFI_PAGE_SIZE;
42 size_t pageCount = Ram / pageSize;
43 size_t pageMapSize = pageCount * sizeof(PageMapEntry);
44 //find a region in the uefi memory map large enough to fit the page map
45 size_t DescriptorSize = UEFI_LEFTOWER::EFI_MEMORY_DESCRIPTOR_SIZE;
46 //size_t memorymapSize = UEFI_LEFTOWER::EFI_MEMORY_MAP_SIZE;
47 size_t PhysicalStartOfMap = 0;
48 size_t PagesUsedByMap = 0;
49
50 EFI_MEMORY_DESCRIPTOR* memoryMap = reinterpret_cast<EFI_MEMORY_DESCRIPTOR*>(UEFI_LEFTOWER::EFI_MEMORY_DESCRIPTOR_);
51 //enumerate the map
52 for (UINTN i = 0; i < UEFI_LEFTOWER::EFI_MEMORY_MAP_SIZE / DescriptorSize; i++) {
53 EFI_MEMORY_DESCRIPTOR* Descriptor = (EFI_MEMORY_DESCRIPTOR*)((UINT8*)memoryMap + i * DescriptorSize);
54 if (Descriptor->Type == EfiConventionalMemory && Descriptor->NumberOfPages * EFI_PAGE_SIZE >= pageMapSize) {
55 PhysicalStartOfMap = Descriptor->PhysicalStart;
56 PagesUsedByMap = pageMapSize / pageSize;
57 break;
58 }
59 }
60 if (PhysicalStartOfMap == 0 || PagesUsedByMap == 0) {
61 return false; // No suitable memory region found
62 }
63 //now that we have a location for the map we can start recording the physical pages into it
64 //this involves treating the found region as an array of PageMapEntry
65 //to make these entrys usefull well enumerate the map again and record allready used regions into the map and recording their type.
66
67 SYSTEM::SYSTEM_INFO::GlobalPageMap.Entrys = reinterpret_cast<PageMapEntry*>(PhysicalStartOfMap);
68 SYSTEM::SYSTEM_INFO::GlobalPageMap.sizeBytes = pageMapSize;
69 SYSTEM::SYSTEM_INFO::GlobalPageMap.size_pages = PagesUsedByMap;
70 SYSTEM::SYSTEM_INFO::GlobalPageMap.etries = pageCount;
71 for (size_t i = 0; i < pageCount; i++) {
72 SYSTEM::SYSTEM_INFO::GlobalPageMap.Entrys[i].physicalStart = PhysicalStartOfMap + i * pageSize;
73 SYSTEM::SYSTEM_INFO::GlobalPageMap.Entrys[i].Type = PageType::Free; // Initialize as free
74 SYSTEM::SYSTEM_INFO::GlobalPageMap.Entrys[i].isFree = true;
75 }
76 //now we will enumerate the map again and record allready used regions into the map and recording their type.
77 for (UINTN i = 0; i < UEFI_LEFTOWER::EFI_MEMORY_MAP_SIZE / DescriptorSize; i++) {
78 EFI_MEMORY_DESCRIPTOR* Descriptor = (EFI_MEMORY_DESCRIPTOR*)((UINT8*)memoryMap + i * DescriptorSize);
79 //base stuff necesary for all memory types
80
81 UINT32 DescriptorType = Descriptor->Type;
82
83 void* PhysicalBaase = reinterpret_cast<void*>(Descriptor->PhysicalStart);
84 size_t numPages = Descriptor->NumberOfPages;
85 size_t FirstPage = reinterpret_cast<uintptr_t>(PhysicalBaase) / pageSize;
86 switch (DescriptorType) {
87 //first all the Reserved Memory stuuf. theyll all falltrough to the same thing to make it easier
88
89 case EfiReservedMemoryType: { __fallthrough; }
90 case EfiRuntimeServicesCode: { __fallthrough; }
91 case EfiRuntimeServicesData: { __fallthrough; }
92 case EfiUnusableMemory: { __fallthrough; }
93 case EfiACPIReclaimMemory: { __fallthrough; }
94 case EfiACPIMemoryNVS: { __fallthrough; }
95 case EfiPalCode: { __fallthrough; }
96 //all of the reserved memory descriptors falltrough to this
97 case EfiPersistentMemory: {
98 //ill itterato trough all the pages and set thepaesin the oage table to be reserved
99 for (size_t page = FirstPage; page < FirstPage + numPages; page++) {
100 PageMapEntry& curr = SYSTEM_INFO::GlobalPageMap.Entrys[page];
101 curr.isFree = false;
102 //curr.physicalStart=PhysicalBaase //not neceary to reassign the physical base as this is done in the step before
103 curr.Type = PageType::EFI_Reserved;// for now i dont differenciate between reserved memory types
104 }
105
106
107 break;
108 }
109 //now il handle the maped io stuff
110 case EfiMemoryMappedIO: { __fallthrough; }
111 case EfiMemoryMappedIOPortSpace: {
112 for (size_t page = FirstPage; page < FirstPage + numPages; page++) {
113
114 PageMapEntry& curr = SYSTEM_INFO::GlobalPageMap.Entrys[page];
115 curr.isFree = false;//obviously mmio isnt free
116 curr.Type = PageType::MMIO;
117 }
118 break;
119 }
120 //and now ill handle usable memory
121 case EfiConventionalMemory: {
122 for (size_t page = FirstPage; page < FirstPage + numPages; page++) {
123
124 PageMapEntry& curr = SYSTEM_INFO::GlobalPageMap.Entrys[page];
125 curr.isFree = true; //this is free space
126 curr.Type = PageType::Free;//and its frre
127 }
128 break;
129 }
130 }
131 }
132 return true;
